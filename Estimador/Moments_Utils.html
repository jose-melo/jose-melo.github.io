import scipy as sci

def getMomentCentered(center, delta, S, f):

    '''
    ### Calculates the moments of a function S(f).
    
    ### It uses the expected frequency to center the frequency spectrum in order to calculate to moments.

      - params:
        1. **center  :**    Expeted value to measure
        2. **delta   :**    The size of each half of the frequency window interval
        3. **S       :**    Input function: signal spectrum
        4. **f       :**    Independent variable: frequency

      - returns:
        1. **m0   :**    Zero order moment
        2. **m2   :**    Second order moment
        3. **Tmin :**    Min period considered in the time window
        4. **Tmax :**    Max period considered in the time window

    '''

    #Timestep used in numerical integration
    Ts = 1

    # [fc1, fc2]: interval considered
    fc1 = center - delta
    fc2 = center + delta

    
    for i in range(len(f)):
        if f[i] >= fc1:
            idx1 = i
            break
    
    for j in range(len(f)-1, 0, -1):
        if f[j] <= fc2:
            idx2 = j
            break
                
    fm2 = []
    for i in range(idx1, idx2):
        fm2.append((f[i]**2)*S[i])

    fm0 = []
    for i in range(idx1, idx2):
        fm0.append(S[i])

        
    #Zero order moment
    m0 = sci.integrate.simps(fm0, dx=Ts)

    #Second order moment
    m2 = sci.integrate.simps(fm2, dx=Ts)

    #Max. period considered
    Tmax = 1/f[idx1]

    #Min period considered
    Tmin = 1/f[idx2]

    return m0, m2, Tmin, Tmax
