def getNaturalPeriod_AutoCorrelation(serie, center):

    '''
      ### Calculates the natural period from serie using the autocorrelation function. 
      
      ### It uses the expected value for the natural period to choose the next one.
      
      ### The time window is fixed to 2 hours. Considering a sampling frequency of 1 Hz (constant throughout the work), the number of points is fixed to 7200.

        - params: 
          1. **serie:** Input temporal series
          2. **center:** Expected value

        - local params:
          1. **N:** Number of points per window. 
          2. **delta:** Steps for the scrolling window (# of displaced points)
          3. **div:** Number of divisions per window. (Used to calculate mean values)

        - returns:
          1. **measuredTp:** Python list with the values of measured Tp for each window
        
    '''

    measuredTp = []
    N = 7200
    delta =  int(0.01*N)
    div = 1

        corr =  autoCorrelation(serie[i*delta:i*delta + N], 300)
        
        #The next steps will transform the domain from period to frequency
        
        #Get the frequency axis
        Oxcorr = np.arange(len(corr))
        Ofreq = []
        for i in Oxcorr:
            Ofreq.append(1/i)
        
        #Auxiliary list
        auxList = []
        for i in range(len(Ofreq)):
            auxList.append([Ofreq[i], corr[i]])
        
        #Sort using the frequencies
        auxList = sorted(auxList, key=lambda x : x[0])
        
        #Get the values in the new domain
        for i in range(len(auxList)):

        #print(np.array(T))
