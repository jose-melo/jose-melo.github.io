import pandas as pd
import numpy as np
import plotly.express as px
from matplotlib import pyplot as plt

def getPosition(graph, arq, plotType):

    '''
     ### Function that returns the time series on the local axes (x, y, z).

     ### Note: the rotation matrix used in the transformation uses the angle Yaw (oriented counterclockwise)

     ### **Yaw angle:** where 0 degree is the situation where the local x-axis of the float coincides with the global X axis, with angles growing positively counterclockwise.

     ### **Heading:** where 0 degree is the situation where the local x-axis of the float coincides with the North (global Y axis), and the angles grow positively clockwise. This system is commonly used by offshore technicians.


       - params:
         1. **graph     :**  Boolean variable that handles printing graphics. True: plot graphics
         2. **file      :**  Positions file. Ex: "v0001.pos"
         3. **plotType  :**  Specifies the type of plot used. 2 for matplotlib plot other value to plotly plot

       - return:
         1. **(x, y, z, Yaw):** local axes of the vessel.

    '''

    dados = levessel(arq)

    X = dados[0]
    Y = dados[1]
    Z = dados[2]
    yaw = (dados[5]*mt.pi)/180

    # Transformation
    x = X*np.cos(yaw) + Y*np.sin(yaw)
    y = -X*np.sin(yaw) + Y*np.cos(yaw)

    
    #Ignore the first points due to transitions effects
    x = x[500:]
    y = y[500:]
    Z = Z[500:]
    yaw = yaw[500:]
    
    if(not graph):
        return x, y, Z, yaw
    
    time = np.arange(0, int(len(x)), 1)
    
    if(plotType == 2):
        plt.figure(figsize=(20, 5))
        plt.plot(time, x)
        plt.xlabel("Time (s)", fontdict={'fontsize':15})
        plt.ylabel("X (m)", fontdict={'fontsize':15})
        plt.title("Time series: X ", fontdict={'fontsize':15})
        plt.plot()

        plt.figure(figsize=(20, 5))
        plt.plot(time, y)
        plt.xlabel("Time (s)", fontdict={'fontsize':15})
        plt.ylabel("Y (m)", fontdict={'fontsize':15})
        plt.title("Time series: Y", fontdict={'fontsize':15})
        plt.show()
        
        plt.figure(figsize=(20, 5))
        plt.plot(time, yaw)
        plt.xlabel("Time (s)", fontdict={'fontsize':15})
        plt.ylabel("Yaw (rad)", fontdict={'fontsize':15})
        plt.title("Time series: Yaw ", fontdict={'fontsize':15})
        plt.show()
        
        return x, y, Z, yaw
    
    df = pd.DataFrame({"Time (s)":time, "X - local (m)":x})
    fig = px.line(df, x='Time (s)', y="X - local (m)", title="Time series: X")
    fig.show()

    df = pd.DataFrame({"Time (s)":time, "Y - local (m)":y})
    fig = px.line(df, x='Time (s)', y="Y - local (m)", log_y=False, title="Time series: Y")
    fig.show()

    df = pd.DataFrame({"Time (s)":time, "Z - local (m)":Z})
    fig = px.line(df, x='Time (s)', y="Z - local (m)", log_y=False, title="Time series: Z")
    fig.show()
    
    df = pd.DataFrame({"Time (s)":time, "Yaw - local (rad)":yaw})
    fig = px.line(df, x='Time (s)', y="Yaw - local (rad)", log_y=False, title="Time series: Yaw")
    fig.show()

    return x, y, Z, yaw

#------------------------------------------------------------------------------
# levessel(filepath)
#
# Reads Positions, Velocities and Accelerations from .pos binary file
#        
# INPUT:
# filepath: String containing the filepath of a binary .pos vessel file with
#           position, velocity and acceleration timeseries.
#        
# OUTPUT:
# data_array: (18,n*dt) numpy array with data read from .pos file
#
# data_array[0:5, n*dt] = desloc x,y,z, rx, ry, rz do navio no instante n*dt
# data_array[6:11, n*dt] = veloc x,y,z, rx, ry, rz do navio no instante n*dt
# data_array[12:17, n*dt] = accel x,y,z, rx, ry, rz do navio no instante n*dt
#------------------------------------------------------------------------------
def levessel(filepath):
    with open(filepath, 'rb') as fid:
        data_array = np.fromfile(fid, np.single).reshape((-1, 18)).T
    
    return data_array

#------------------------------------------------------------------------------
# levefor(filepath)
#
# Reads Forces from .for binary file
#        
# INPUT:
# filepath: String containing the filepath of a binary .for file with
#           forces timeseries.
#        
# OUTPUT:
# data_array: (96,n*dt) numpy array with data read from .for file
#
# 6 force components for each action below:
# * 1 Inertial - (X,Y,Z, XX, YY, ZZ);
# * 2 Current(drag) - (X,Y,Z, XX, YY, ZZ);
# * 3 Wind - (X,Y,Z, XX, YY, ZZ);
# * 4 First Order(wave) - (X,Y,Z, XX, YY, ZZ);
# * 5 Mean Drift(wave) - (X,Y,Z, XX, YY, ZZ);
# * 6 Slow Drift(wave) - (X,Y,Z, XX, YY, ZZ);
# * 7 Wave Drift Damping(wave) - (X,Y,Z, XX, YY, ZZ);
# * 8 First Order(swell) - (X,Y,Z, XX, YY, ZZ);
# * 9 Mean Drift(swell) - (X,Y,Z, XX, YY, ZZ);
# *10 Slow Drift(swell) - (X,Y,Z, XX, YY, ZZ);
# *11 Swell Drift Damping(swell) - (X,Y,Z, XX, YY, ZZ);
# *12 Damping (total) - (X,Y,Z, XX, YY, ZZ);
# *13 Hydrostatic(restoration) - (X,Y,Z, XX, YY, ZZ);
# *14 Mooring(All lines) - (X,Y,Z, XX, YY, ZZ);
# *15 Line damp Drag - (X,Y,Z, XX, YY, ZZ);
# *16 Dynamic Positioning - (X,Y,Z, XX, YY, ZZ);
# *17 Total - (X,Y,Z, XX, YY, ZZ).
#------------------------------------------------------------------------------
def levefor(filepath):
    with open(filepath, 'rb') as fid:
        data_array = np.fromfile(fid, np.single).reshape((-1, 102)).T
    
    return data_array
