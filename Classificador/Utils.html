import numpy as np
from matplotlib import pyplot as plt

def plot_cmatrix(matrix,      class_names,
                          title='Confusion matrix',
                          gradient=None,
                          proportions=True):
    """
    It constructs and plots the confusion matrix

    Arguments
    ---------
    matrix:      Input matrix with prediction results

    class_names: Name of the classes, ex: ['L01', 'L05', 'L09', 'no_failure']

    title:       Title of the plot

    gradient:    Python colormap

    proportions:  True plot the proportions, False plot the raw values
    
    """

    acc = np.trace(matrix) / float(np.sum(matrix))
    misclass = 1 - acc

    if gradient is None:
        gradient = plt.get_gradient('Blues')

    plt.figure(figsize=(8, 6))
    plt.imshow(matrix, interpolation='nearest', cmap=gradient)
    plt.title(title)
    plt.colorbar()

    if class_names is not None:
        tick_marks = np.arange(len(class_names))
        plt.xticks(tick_marks, class_names, rotation=45)
        plt.yticks(tick_marks, class_names)

    if proportions:
        matrix = (matrix.astype('float')*100) / matrix.sum(axis=1)[:, np.newaxis]


    thresh = matrix.max() / 1.5 if proportions else matrix.max() / 2
    for i, j in itertools.product(range(matrix.shape[0]), range(matrix.shape[1])):
        if proportions:
            plt.text(j, i, "{:0.2f}%".format(matrix[i, j]),
                     horizontalalignment="center",
                     color="white" if matrix[i, j] > thresh else "black")
        else:
            plt.text(j, i, "{:,}".format(matrix[i, j]),
                     horizontalalignment="center",
                     color="white" if matrix[i, j] > thresh else "black")


    plt.tight_layout()
    plt.ylabel('True label')
    plt.xlabel('Predicted label\nacc={:0.4f}; misclass={:0.4f}'.format(acc, misclass))
    plt.show()
