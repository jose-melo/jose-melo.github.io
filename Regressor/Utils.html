import os

def loadValues(txt):
    
    '''
        ### Reads the values of period and position. These files were generated for the above function.
        
          - params:
            1. **path:** Path of the folders.
            2. **txt:**  Name of the file to read. Also specifies the value.
        
        ### Name of the files:
            1. offX.txt
            2. offY.txt
            3. offYaw.txt
            4. stdX.txt
            5. stdY.txt
            6. stdZ.txt
            7. stdRoll.txt
            8. stdPitch.txt
            9. stdYaw.txt
            10. TX.txt
            11. TY.txt
            12 .TYaw.txt
            
    '''
    
    lista = []
    path = "saves"
    os.chdir(path)
    arq = open(txt, 'r')
    linhas = arq.readlines()
    arq.close()
    for i in linhas:
        lista.append(float(i))
    
    os.chdir('..')
    
    path = "saves-1001-2900"
    os.chdir(path)
    arq = open(txt, 'r')
    linhas = arq.readlines()
    arq.close()
    for i in linhas:
        lista.append(float(i))

    os.chdir('..')
    
    return lista

def carrega2(txt, path):
    '''
        ### Reads a .txt file when a single path is required.

          - params:
            1. **txt :** File name
            2. **path:** File path
 
          - returns:
            1. **list:** Python list with the data
    '''


    lista = []
    os.chdir(path)
    try:
        arq = open(txt, 'r')
        linhas = arq.readlines()
        arq.close()
        for i in linhas:
            lista.append(float(i))
    except Exception as e:
        
        print(e)
        
    os.chdir('..')
    return lista
